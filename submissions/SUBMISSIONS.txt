## Unified experiment config (matches the M1 feedback on "matched hyperparameters")
# All 7 runs (2 baselines + 5 custom) share:
#   loss      = NSSALoss(margin=9.0, adversarial_temperature=1.0)   # RotatE paper default
#   sampler   = Bernoulli with filtered=True
#   num_negs  = 8 negatives per positive
#   model     = embedding_dim=128, Adam lr=1e-3, batch_size=1024
#   training  = 50 epochs max, early stopping on validation MRR, patience=10
#   seed      = 42
# Only what changes between runs: model architecture (TransE vs RotatE) and the
# candidate-selection strategy (PyKEEN default Bernoulli for baselines vs
# random / hard / mixed selection from a scored candidate pool for the custom runs).

## C) Smoke tests (run before any long training, ~30s each)
  # Baseline (TransE 1 epoch on a small batch)
  python code/train_baseline_kge.py --model TransE --epochs 1 --batch_size 256 --patience 1

  # Custom pipeline (mixed 1 epoch, 10 batches, 200 val triples)
  python code/train_rotate_custom.py --strategy mixed --hard-fraction 0.5 \

    --limit-batches 10 --limit-val-eval 200

## D) Baseline runs (2 runs, ~50 min total on GPU)
  python code/train_baseline_kge.py --model TransE --epochs 50 --batch_size 1024 --patience 10
  python code/train_baseline_kge.py --model RotatE --epochs 50 --batch_size 1024 --patience 10
  # Artifacts: artifacts/baseline/{TransE,RotatE}_summary.txt, *_curves.png, pykeen_{TransE,RotatE}/

## E) Custom RotatE runs (5 runs, ~4-6 h total on GPU; do NOT run `git pull` between them)
  # Main 3 strategies (the core of the D3 / DL-final claim)
  python code/train_rotate_custom.py --strategy random --num-negs 8 --epochs 50 --patience 10
  python code/train_rotate_custom.py --strategy hard   --num-negs 8 --epochs 50 --patience 10
  python code/train_rotate_custom.py --strategy mixed --hard-fraction 0.5 --num-negs 8 --epochs 50 --patience 10
  # Hard-fraction ablation
  python code/train_rotate_custom.py --strategy mixed --hard-fraction 0.3 --num-negs 8 --epochs 50 --patience 10
  python code/train_rotate_custom.py --strategy mixed --hard-fraction 0.7 --num-negs 8 --epochs 50 --patience 10
  # Artifacts: artifacts/custom/RotatE_{random,hard,mixed_30_70,mixed_50_50,mixed_70_30}/
  #   summary.txt, history.json, curves.png, trained_model.pkl

## F) Slice + qualitative evaluation (after all runs complete)
  python code/evaluate_slices.py --all
  # Produces artifacts/<baseline|custom>/<run>/slices.json with MRR/H@10 by:
  #   - relation-frequency bucket (rare / mid / frequent) computed on train only
  #   - entity-degree bucket (low / mid / high) computed on train only
  #   - head vs tail prediction separately

  python code/qualitative_examples.py --num-triples 5
  # Produces artifacts/custom/qualitative.md with top-10 predictions per model.
